class Solution {
public:
    int maxArea(vector<int>& height) {
        int maxx=0;
        int left=0;
        int right=height.size()-1;
        while(left<right){
            int wid=right-left;
            int hei=min(height[left],height[right]);
            maxx=max(maxx,wid*hei);
            if(height[left]<height[right]){
                left++;
            }
            else{
                right--;
            }
        }
        return maxx;
    }
};
