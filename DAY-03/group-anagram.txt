class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        unordered_map<string,vector<string>>mp;
        int n=strs.size();
        for(int i=0;i<n;i++){
            string cop=strs[i];
            sort(cop.begin(),cop.end());
            mp[cop].push_back(strs[i]);
        }
        vector<vector<string>>vec;
        for(auto it:mp){
            vec.push_back(it.second);
        }
        return vec;
    }
};
