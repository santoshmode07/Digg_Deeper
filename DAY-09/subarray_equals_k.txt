class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
    int n=nums.size(),s=0,count=0;
    unordered_map<int,int>mp;
    if(n<1){
        return 0;
    }
    mp[0]=1;
    for(int i=0;i<n;i++){
        s+=nums[i];
        if(mp.find(s-k)!=mp.end()){
            count=count+mp[s-k];
        }
        mp[s]++;
    }
    return count;
    }
};
