class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        int n=nums.size();
        int left=0,right=n-1;
        while(left<=right){
            int mid=(left+right)/2;
            int summ=nums[left]+nums[right];
            if(summ==target){
                return {left+1,right+1};
            }
            else if(summ<target){
                left++;
            }
            else{
                right--;
            }
        }
        return {};
    }
};
