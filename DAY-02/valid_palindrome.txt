    class Solution {
public:
    bool isPalindrome(string s) {
        string ss;
        for(char a:s){
            if(isalnum(a))
            {
                ss+=tolower(a);
            }
        }
        for(int i=0;i<ss.size()/2;i++)
        {
            if(ss[i]!=ss[ss.size()-i-1])
            {
                return false;
            }
        }
        return true;
    }
};
