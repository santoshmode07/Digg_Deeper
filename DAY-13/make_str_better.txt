class Solution {
public:
    string makeGood(string s) {
        string ans;
        for(char i:s){
            if(!ans.empty()){
                if(ans.back()==i+32||ans.back()==i-32){
                    ans.pop_back();
                }
                else{
                    ans+=i;
                }
            }
            else{
                ans+=i;
            }
        }
        return ans;
    }
};
