class Solution {
public:
    int climbStairs(int n) {
          if(n==1) return 1;
        if(n==2) return 2;
        int prev_1=1,prev_2=2,curr;
        for(int i=3;i<=n;i++)
        {
            curr=prev_1+prev_2;
            prev_1=prev_2;
            prev_2=curr;
        }
        return prev_2;
    }
};
