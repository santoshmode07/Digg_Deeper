class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        priority_queue<int>pq;
        for(int i=0;i<nums.size();i++){
            pq.push(nums[i]);
        }
        int m=k-1;
        while(m>0){
            pq.pop();
            m--;
        }
        return pq.top();
    }
};
