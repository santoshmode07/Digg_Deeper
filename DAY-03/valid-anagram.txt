class Solution {
public:
    bool isAnagram(string s, string t) {
        int n=s.size(),nn=t.size();
        if(n!=nn){
            return false;
        }
        unordered_map<char,int>mp;
        for(int i=0;i<n;i++){
            mp[s[i]]++;
        }
        unordered_map<char,int>mpp;
        for(int i=0;i<nn;i++){
            mpp[t[i]]++;
        }
        if(mp!=mpp){
            return false;
        }
        return true;
    }
};
